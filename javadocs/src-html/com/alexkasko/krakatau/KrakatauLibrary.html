006    import org.python.util.PythonInterpreter;

008    import javax.tools.JavaCompiler;

015    import static org.apache.commons.io.FileUtils.listFiles;

017    /**
018     * Frontend library for java classfiles operations: disassembling, assembling, decompilation and compilation.
019     * Uses eclipse compiler for compilation and krakatau python library (through jython) for other operations
020     *
021     * @author alexkasko
022     * Date: 9/29/13
023     */
024    public class KrakatauLibrary {
025        private static final Charset UTF8 = Charset.forName("UTF-8");
026    
027        private final PythonInterpreter python;
028    
029        /**
030         * Constructor
031         */
032        public KrakatauLibrary() {
033    //        http://bugs.jython.org/issue1435
034    //        Options.showJavaExceptions = true;
035    //        Options.includeJavaStackInExceptions = true;
036            this.python = new PythonInterpreter();
037        }
038    
039        /**
040         * Decompiles list of fully qualified classes from specified classpath
041         *
042         * @param classpathFiles list of classpath entries (.jar files or directories)
043         * @param fqClassNames list of fully qualifies class names
044         * @param outDir output directory
045         * @throws KrakatauException
046         */
047        public void decompile(Collection<File> classpathFiles, Collection<String> fqClassNames, File outDir) throws KrakatauException {
048            if(null == classpathFiles) throw new KrakatauException("Specified 'classpathFiles' is null");
049            if(null == fqClassNames) throw new KrakatauException("Specified 'fqClassNames' is null");
050            if(0 == fqClassNames.size()) throw new KrakatauException("Specified 'fqClassNames' is empty");
051            if(null == outDir) throw new KrakatauException("Specified 'outDir' is null");
052            if(!(outDir.exists() && outDir.isDirectory())) throw new KrakatauException("Invalid output directory: [" + outDir.getAbsolutePath() + "]");
053            // find rt.jar
054            String jrePath = System.getProperty("java.home");
055            File jreDir = new File(jrePath);
056            if(!(jreDir.exists() && jreDir.isDirectory())) throw new KrakatauException(
057                    "Invalid JRE dir: [" + jreDir.getAbsolutePath() +"] obtained through 'java.home' property");
058            File rtJar = new File(jreDir, "lib/rt.jar");
059            if(!(rtJar.exists() && rtJar.isFile())) throw new KrakatauException(
060                    "Cannot access 'rt.jar' on path: [" + rtJar.getAbsolutePath() + "]");
061            // prepare classpath
062            List<String> classpath = new ArrayList<String>(classpathFiles.size() + 1);
063            classpath.add(rtJar.getPath());
064            for(File fi : classpathFiles) {
065                classpath.add(fi.getPath());
066            }
067            String classpathStr = toPythonList(classpath);
068            // prepare files
069            List<String> classes = new ArrayList<String>(fqClassNames.size());
070            for(String cl : fqClassNames) {
071                String stripped = cl.endsWith(".class") ? cl.substring(0, cl.length() - 6) : cl;
072                String name = stripped.replace(".", "/");
073                classes.add(name);
074            }
075            String classesStr = toPythonList(classes);
076            // run krakatau
077            try {
078                python.exec("import decompile");
079                python.exec("decompile.decompileClass(" + classpathStr + ", " + classesStr + ", '" + outDir.getPath() + "')");
080            } catch (Exception e) {
081                throw new KrakatauException("Decompile error", e);
082            }
083        }
084    
085        /**
086         * Disassembles list of class files (or directories) into asm (.j) files
087         *
088         * @param classSources list of class files (or directories)
089         * @param outDir output directory
090         * @throws KrakatauException
091         */
092        public void disassemble(Collection<File> classSources, File outDir) throws KrakatauException {
093            if(null == classSources) throw new KrakatauException("Specified 'classSources' is null");
094            if(0 == classSources.size()) throw new KrakatauException("Specified 'classSources' is empty");
095            if(null == outDir) throw new KrakatauException("Specified 'outDir' is null");
096            if(!(outDir.exists() && outDir.isDirectory())) throw new KrakatauException("Invalid output directory: [" + outDir.getAbsolutePath() + "]");
097            // prepare files
098            List<File> classFiles = expandDirs(classSources, "class");
099            List<String> paths = new ArrayList<String>(classFiles.size());
100            for (File fi : classFiles) {
101                paths.add(fi.getPath());
102            }
103            String pathsStr = toPythonList(paths);
104            // run krakatau
105            try {
106                python.exec("import disassemble");
107                python.exec("disassemble.disassembleClass(disassemble.readFile, " + pathsStr + ", '" + outDir.getPath() + "')");
108            } catch (Exception e) {
109                throw new KrakatauException("Disassemble error", e);
110            }
111        }
112    
113        /**
114         * Assembled list of asm (.j) files (or directories) into class files
115         *
116         * @param asmSources list of asm files (or directories)
117         * @param outDir output directory
118         * @throws KrakatauException
119         */
120        public void assemble(Collection<File> asmSources, File outDir) throws KrakatauException {
121            if(null == asmSources) throw new KrakatauException("Specified 'asmSources' is null");
122            if(0 == asmSources.size()) throw new KrakatauException("Specified 'asmSources' is empty");
123            if(null == outDir) throw new KrakatauException("Specified 'outDir' is null");
124            if(!(outDir.exists() && outDir.isDirectory())) throw new KrakatauException("Invalid output directory: [" + outDir.getAbsolutePath() + "]");
125            List<File> asmFiles = expandDirs(asmSources, "j");
126            // run krakatau
127            try {
128                python.exec("import assemble");
129                python.exec("from Krakatau import script_util");
130                for (File fi : asmFiles) {
131                    python.exec("pairs = assemble.assembleClass('" + fi.getPath() + "', True, False)");
132                    python.exec("for name, data in pairs:\n" +
133                            "    filename = script_util.writeFile('" + outDir.getPath() + "', name, '.class', data)\n" +
134                            "    print 'Class written to', filename");
135                }
136            } catch (Exception e) {
137                throw new KrakatauException("Assemble error", e);
138            }
139        }

141        /**
142         * Compiles list of source files (or directories)
143         *
144         * @param sources list of source files (or directories)
145         * @param classpathFiles list of classpath entries (.jar files or directories)
146         * @param outDir output directory
147         * @param errorWriter error writer
148         * @throws KrakatauException
149         */
150        public void compile(List<File> sources, Collection<File> classpathFiles, File outDir, Writer errorWriter) throws KrakatauException {
151            if(null == sources) throw new KrakatauException("Specified 'sources' is null");
152            if(0 == sources.size()) throw new KrakatauException("Specified 'sources' is empty");
153            if(null == classpathFiles) throw new KrakatauException("Specified 'classpathFiles' is null");
154            if(null == outDir) throw new KrakatauException("Specified 'outDir' is null");
155            if(!(outDir.exists() && outDir.isDirectory())) throw new KrakatauException("Invalid output directory: [" + outDir.getAbsolutePath() + "]");
156            // options
157            List<String> options = new ArrayList<String>();
158            // debug
159            options.add("-g");
160            // classpath
161            if (classpathFiles.size() > 0) {
162                boolean first = true;
163                StringBuilder cpBuilder = new StringBuilder();
164                for (File fi : classpathFiles) {
165                    if (first) first = false;
166                    else cpBuilder.append(":");
167                    cpBuilder.append(fi.getPath());
168                }
169                options.add("-classpath");
170                options.add(cpBuilder.toString());
171            }
172            // out
173            options.add("-d");
174            options.add(outDir.getPath());
175            // sources
176            List<File> srcFiles = expandDirs(sources, "java");
177            List<EclipseFileObject> compilationUnits = new ArrayList<EclipseFileObject>(srcFiles.size());
178            for(File fi : srcFiles) {
179                compilationUnits.add(new EclipseFileObject(null, fi.toURI(), JavaFileObject.Kind.SOURCE, UTF8));
180            }
181            // run compiler
182            try {
183                JavaCompiler compiler = new EclipseCompiler();
184                JavaCompiler.CompilationTask compile = compiler.getTask(errorWriter, new EclipseFileManager(Locale.ENGLISH, UTF8),
185                        null, options, null, compilationUnits);
186                compile.call();
187            } catch (Exception e) {
188                throw new KrakatauException("Compile error", e);
189            }
190        }
191    
192        private String toPythonList(Collection<String> col) {
193            StringBuilder sb = new StringBuilder();
194            sb.append("[");
195            boolean first = true;
196            for(String st : col) {
197                if (first) first = false;
198                else sb.append(", ");
199                sb.append("'");
200                sb.append(st);
201                sb.append("'");
202            }
203            sb.append("]");
204            return sb.toString();
205        }
206    
207        private List<File> expandDirs(Collection<File> filesOrDirs, String extension) {
208            List<File> files = new ArrayList<File>();
209            for (File src : filesOrDirs) {
210                if (!src.exists()) throw new KrakatauException("Invalid file or directory: [" + src.getAbsolutePath() + "]");
211                if (src.isFile()) {
212                    files.add(src);
213                } else {
214                    for (File fi : listFiles(src, new String[]{extension}, true)) {
215                        files.add(fi);
216                    }
217                }
218            }
219            return files;
220        }
