018/**
019 * Frontend library for java classfiles operations: disassembling, assembling, decompilation and compilation.

022 * @author alexkasko

025public class KrakatauLibrary {
026    private static final Charset UTF8 = Charset.forName("UTF-8");

028    private final PythonInterpreter python;

049    /**
050     * Decompiles list of fully qualified classes from specified classpath
051     *
052     * @param classpathFiles list of classpath entries (.jar files or directories)
053     * @param fqClassNames list of fully qualifies class names
054     * @param outDir output directory
055     * @throws KrakatauException
056     */
057    public void decompile(Collection<File> classpathFiles, Collection<String> fqClassNames, File outDir) throws KrakatauException {
058        if(null == classpathFiles) throw new KrakatauException("Specified 'classpathFiles' is null");
059        if(null == fqClassNames) throw new KrakatauException("Specified 'fqClassNames' is null");
060        if(0 == fqClassNames.size()) throw new KrakatauException("Specified 'fqClassNames' is empty");
061        if(null == outDir) throw new KrakatauException("Specified 'outDir' is null");
062        if(!(outDir.exists() && outDir.isDirectory())) throw new KrakatauException("Invalid output directory: [" + outDir.getAbsolutePath() + "]");
063        // find rt.jar
064        String jrePath = System.getProperty("java.home");
065        File jreDir = new File(jrePath);
066        if(!(jreDir.exists() && jreDir.isDirectory())) throw new KrakatauException(
067                "Invalid JRE dir: [" + jreDir.getAbsolutePath() +"] obtained through 'java.home' property");
068        File rtJar = new File(jreDir, "lib/rt.jar");
069        if(!(rtJar.exists() && rtJar.isFile())) throw new KrakatauException(
070                "Cannot access 'rt.jar' on path: [" + rtJar.getAbsolutePath() + "]");
071        // prepare classpath
072        List<String> classpath = new ArrayList<String>(classpathFiles.size() + 1);
073        classpath.add(rtJar.getPath());
074        for(File fi : classpathFiles) {
075            classpath.add(fi.getPath());
076        }
077        String classpathStr = toPythonList(classpath);
078        // prepare files
079        List<String> classes = new ArrayList<String>(fqClassNames.size());
080        for(String cl : fqClassNames) {
081            String stripped = cl.endsWith(".class") ? cl.substring(0, cl.length() - 6) : cl;
082            String name = stripped.replace(".", "/");
083            classes.add(name);
084        }
085        String classesStr = toPythonList(classes);
086        // run krakatau
087        try {
088            python.exec("import decompile");
089            python.exec("decompile.decompileClass(" + classpathStr + ", " + classesStr + ", '" + outDir.getPath() + "')");
090        } catch (Exception e) {
091            throw new KrakatauException("Decompile error", e);
092        }
093    }

095    /**
096     * Disassembles list of class files (or directories) into asm (.j) files
097     *
098     * @param classSources list of class files (or directories)
099     * @param outDir output directory
100     * @throws KrakatauException
101     */
102    public void disassemble(Collection<File> classSources, File outDir) throws KrakatauException {
103        if(null == classSources) throw new KrakatauException("Specified 'classSources' is null");
104        if(0 == classSources.size()) throw new KrakatauException("Specified 'classSources' is empty");
105        if(null == outDir) throw new KrakatauException("Specified 'outDir' is null");
106        if(!(outDir.exists() && outDir.isDirectory())) throw new KrakatauException("Invalid output directory: [" + outDir.getAbsolutePath() + "]");
107        // prepare files
108        List<File> classFiles = expandDirs(classSources, "class");
109        List<String> paths = new ArrayList<String>(classFiles.size());
110        for (File fi : classFiles) {
111            paths.add(fi.getPath());
112        }
113        String pathsStr = toPythonList(paths);
114        // run krakatau
115        try {
116            python.exec("import disassemble");
117            python.exec("disassemble.disassembleClass(disassemble.readFile, " + pathsStr + ", '" + outDir.getPath() + "')");
118        } catch (Exception e) {
119            throw new KrakatauException("Disassemble error", e);
120        }
121    }
122
123    /**
124     * Assembled list of asm (.j) files (or directories) into class files
125     *
126     * @param asmSources list of asm files (or directories)
127     * @param outDir output directory
128     * @throws KrakatauException
129     */
130    public void assemble(Collection<File> asmSources, File outDir) throws KrakatauException {
131        if(null == asmSources) throw new KrakatauException("Specified 'asmSources' is null");
132        if(0 == asmSources.size()) throw new KrakatauException("Specified 'asmSources' is empty");
133        if(null == outDir) throw new KrakatauException("Specified 'outDir' is null");
134        if(!(outDir.exists() && outDir.isDirectory())) throw new KrakatauException("Invalid output directory: [" + outDir.getAbsolutePath() + "]");
135        List<File> asmFiles = expandDirs(asmSources, "j");
136        // run krakatau
137        try {
138            python.exec("import assemble");
139            python.exec("from Krakatau import script_util");
140            for (File fi : asmFiles) {
141                python.exec("pairs = assemble.assembleClass('" + fi.getPath() + "', True, False)");
142                python.exec("out = script_util.makeWriter('" + outDir.getPath() + "', '.class')");
143                python.exec("for name, data in pairs:\n" +
144                        "    filename = out.write(name, data)\n" +
145                        "    print 'Class written to', filename");
146            }
147        } catch (Exception e) {
148            throw new KrakatauException("Assemble error", e);
149        }
150    }

152    /**
153     * Compiles list of source files (or directories)
154     *
155     * @param sources list of source files (or directories)
156     * @param langLevel java language and bytecode compliance level (3 - 8)
157     * @param classpathFiles list of classpath entries (.jar files or directories)
158     * @param outDir output directory
159     * @param errorWriter error writer
160     * @throws KrakatauException
161     */
162    public void compile(List<File> sources, int langLevel, Collection<File> classpathFiles, File outDir, Writer errorWriter) throws KrakatauException {
163        if(null == sources) throw new KrakatauException("Specified 'sources' is null");
164        if(langLevel < 3 || langLevel > 8) throw new KrakatauException("Specified 'langLevel': [" + langLevel + "]" +
165                " not supported, supported levels are from [3] to [7]");
166        if(0 == sources.size()) throw new KrakatauException("Specified 'sources' is empty");
167        if(null == classpathFiles) throw new KrakatauException("Specified 'classpathFiles' is null");
168        if(null == outDir) throw new KrakatauException("Specified 'outDir' is null");
169        if(!(outDir.exists() && outDir.isDirectory())) throw new KrakatauException("Invalid output directory: [" + outDir.getAbsolutePath() + "]");
170        // options
171        List<String> options = new ArrayList<String>();
172        // debug
173        options.add("-g");
174        options.add("-1." + langLevel);
175        // classpath
176        if (classpathFiles.size() > 0) {
177            boolean first = true;
178            StringBuilder cpBuilder = new StringBuilder();
179            for (File fi : classpathFiles) {
180                if (first) first = false;
181                else cpBuilder.append(":");
182                cpBuilder.append(fi.getPath());
183            }
184            options.add("-classpath");
185            options.add(cpBuilder.toString());
186        }
187        // out
188        options.add("-d");
189        options.add(outDir.getPath());
190        // sources
191        List<File> srcFiles = expandDirs(sources, "java");
192        List<EclipseFileObject> compilationUnits = new ArrayList<EclipseFileObject>(srcFiles.size());
193        for(File fi : srcFiles) {
194            compilationUnits.add(new EclipseFileObject(null, fi.toURI(), JavaFileObject.Kind.SOURCE, UTF8));
195        }
196        // run compiler
197        try {
198            JavaCompiler compiler = new EclipseCompiler();
199            JavaCompiler.CompilationTask compile = compiler.getTask(errorWriter, new EclipseFileManager(Locale.ENGLISH, UTF8),
200                    null, options, null, compilationUnits);
201            compile.call();
202        } catch (Exception e) {
203            throw new KrakatauException("Compile error", e);
204        }
205    }
206
207    /**
208     * 1.0 compatibility method, defaults langLevel to 8
209     *
210     * @param sources list of source files (or directories)
211     * @param classpathFiles list of classpath entries (.jar files or directories)
212     * @param outDir output directory
213     * @param errorWriter error writer
214     * @throws KrakatauException
215     */
216    public void compile(List<File> sources, Collection<File> classpathFiles, File outDir, Writer errorWriter) throws KrakatauException {
217        compile(sources, 8, classpathFiles, outDir, errorWriter);
218    }
219
220    private String toPythonList(Collection<String> col) {
221        StringBuilder sb = new StringBuilder();
222        sb.append("[");
223        boolean first = true;
224        for(String st : col) {
225            if (first) first = false;
226            else sb.append(", ");
227            sb.append("'");
228            sb.append(st);
229            sb.append("'");
230        }
231        sb.append("]");
232        return sb.toString();
233    }
234
235    private List<File> expandDirs(Collection<File> filesOrDirs, String extension) {
236        List<File> files = new ArrayList<File>();
237        for (File src : filesOrDirs) {
238            if (!src.exists()) throw new KrakatauException("Invalid file or directory: [" + src.getAbsolutePath() + "]");
239            if (src.isFile()) {
240                files.add(src);
241            } else {
242                for (File fi : listFiles(src, new String[]{extension}, true)) {
243                    files.add(fi);
244                }
245            }
246        }
247        return files;
248    }
